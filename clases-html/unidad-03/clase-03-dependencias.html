Conceptos Te√≥ricos

¬øQu√© es Inyecci√≥n de Dependencias?

Inyecci√≥n de Dependencias (DI) es un patr√≥n donde las dependencias de un componente se proveen externamente en lugar de crearse internamente. Esto permite:

  • Mayor flexibilidad: Cambiar implementaciones sin modificar c√≥digo
  • Mejor testing: Inyectar mocks en lugar de implementaciones reales
  • Menos acoplamiento: Componentes dependen de abstracciones, no implementaciones
  • C√≥digo m√°s limpio: Separaci√≥n de responsabilidades
Analog√≠a: En lugar de que un tel√©fono construya su propia c√°mara, recibe una c√°mara inyectada que puede cambiar entre diferentes modelos.

Inyecci√≥n de Dependencias en FastAPI

FastAPI implementa DI usando la funci√≥n Depends(), que permite declarar dependencias que FastAPI resuelve autom√°ticamente antes de ejecutar el endpoint.

Flujo de Inyecci√≥n

1Ô∏è‚É£ Request llega al endpoint
2Ô∏è‚É£ FastAPI detecta Depends()
3Ô∏è‚É£ Ejecuta funci√≥n de dependencia
4Ô∏è‚É£ Pasa resultado al endpoint
5Ô∏è‚É£ Endpoint usa la dependencia

Ventajas de Depends() en FastAPI

Reutilizaci√≥n

Una funci√≥n de dependencia puede usarse en m√∫ltiples endpoints sin duplicar c√≥digo.

Testing F√°cil

Sobreescribe dependencias con mocks para testing aislado.

Documentaci√≥n Auto

Las dependencias se documentan autom√°ticamente en OpenAPI.

Optimizaci√≥n

FastAPI optimiza llamadas a dependencias compartidas.

def obtener_db():
    """
    Dependencia con setup y teardown.

    yield separa el setup (antes) del teardown (despu√©s).
    """
    # SETUP: Crear conexi√≥n
    db = {"conexion": "bd", "estado": "conectado"}

    try:

        db["estado"] = "desconectado"
        print("Conexi√≥n cerrada")

@app.get("/datos")
def leer_datos(db: dict = Depends(obtener_db)):
    """Usa la BD con limpieza autom√°tica."""
    return {"datos": "...", "estado_db": db["estado"]}

    username: str,
    email: str,
    db: Session = Depends(get_db)

    """Crea usuario en BD usando la sesi√≥n inyectada."""

    db.commit()
    db.refresh(nuevo_usuario)

def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db)

    """
    Verifica el token JWT y retorna el usuario autenticado.

    Args:
        credentials: Credenciales Bearer token extra√≠das del header
        db: Sesi√≥n de BD para buscar usuario

    Returns:
        Usuario: Usuario autenticado

    Raises:
        HTTPException: Si el token es inv√°lido o el usuario no existe
    """
    token = credentials.credentials

    # TODO: Validar token JWT (clase siguiente)
    # Por ahora, simulamos que el token es el username
    username = token  # Simplificaci√≥n

    # Buscar usuario en BD
    usuario = db.query(Usuario).filter(Usuario.username == username).first()
    if not usuario:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Token inv√°lido",
            headers={"WWW-Authenticate": "Bearer"},
        )

    if not usuario.activo:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Usuario inactivo"

    return usuario

from enum import Enum

class Rol(str, Enum):
    """Roles disponibles en el sistema."""
    ADMIN = "admin"
    USUARIO = "usuario"
    INVITADO = "invitado"

def require_rol(rol_requerido: Rol):
    """
    Factory que crea una dependencia que verifica rol.

    Args:
        rol_requerido: Rol necesario para acceder

    Returns:
        Funci√≥n de dependencia
    """
    def dependencia_rol(current_user: Usuario = Depends(get_current_user)) -> Usuario:
        if current_user.rol != rol_requerido:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail="Permiso insuficiente"
            )
        return current_user

    return dependencia_rol

@app.get("/admin/panel", dependencies=[Depends(require_rol(Rol.ADMIN))])

    """Solo administradores pueden acceder."""
    return {"mensaje": "Panel de administraci√≥n"}

DO's con Depends()

  • Usa Depends para l√≥gica reutilizable: Autenticaci√≥n, BD, configuraci√≥n
  • Usa yield para recursos con cleanup: Conexiones, archivos, transacciones
  • Documenta con docstrings: Explica qu√© hace la dependencia
  • Usa type hints: Mejora autocompletado y documentaci√≥n
  • Override en tests: Usa dependency_overrides para testing

DON'Ts con Depends()

  • Usa Depends(get_db)
  • Solo autenticaci√≥n/provisi√≥n
  • Siempre especifica qu√© retorna
  • Rompe testing y mantenibilidad

# dependencies/auth.py

from fastapi import Depends, HTTPException, status
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from sqlalchemy.orm import Session
from typing import Optional

from dependencies.database import get_db
from models.usuario import Usuario

security = HTTPBearer()

def get_current_user(
    credentials: HTTPAuthorizationCredentials = Depends(security),
    db: Session = Depends(get_db)
) -> Usuario:
    """
    Verifica autenticaci√≥n y retorna usuario actual.

    Args:
        credentials: Token Bearer del header Authorization
        db: Sesi√≥n de BD para buscar usuario

    Returns:
        Usuario: Usuario autenticado

    Raises:
        HTTPException: 401 si no autenticado
    """
    # Por ahora, el token es el username (simplificaci√≥n)
    # En la siguiente clase veremos JWT real
    token = credentials.credentials

    # Buscar usuario
    usuario = db.query(Usuario).filter(
        Usuario.username == token
    ).first()

    if not usuario:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Token inv√°lido",
            headers={"WWW-Authenticate": "Bearer"},
        )

    if not usuario.activo:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="Usuario inactivo"
        )

    return usuario

# main.py

from fastapi import FastAPI, Depends
from pydantic import BaseModel
from typing import List
from datetime import datetime

from dependencies.auth import get_current_user
from dependencies.database import get_db
from models.usuario import Usuario
from sqlalchemy.orm import Session

app = FastAPI()

class UsuarioResponse(BaseModel):
    id: int
    username: str
    email: str
    activo: bool

@app.get("/usuarios/me", response_model=UsuarioResponse)
def leer_mi_perfil(
    current_user: Usuario = Depends(get_current_user)
):
    """
    Endpoint protegido que retorna el perfil del usuario autenticado.

    Si no hay token o es inv√°lido, retorna 401 autom√°ticamente.
    """
    return UsuarioResponse(
        id=current_user.id,
        username=current_user.username,
        email=current_user.email,
        activo=current_user.activo
    )

@app.get("/usuarios/mis-proyectos", response_model=List[dict])
def listar_mis_proyectos(
    current_user: Usuario = Depends(get_current_user),
    db: Session = Depends(get_db)
):
    """
    Lista proyectos del usuario autenticado.

    Usa dos dependencias:
    1. get_current_user: Verifica autenticaci√≥n
    2. get_db: Provee sesi√≥n de BD
    """
    from models.proyecto import Proyecto

    proyectos = db.query(Proyecto).filter(
        Proyecto.usuario_id == current_user.id
    ).all()

    return [
        {
            "id": p.id,
            "nombre": p.nombre,
            "descripcion": p.descripcion
        }
        for p in proyectos
    ]
