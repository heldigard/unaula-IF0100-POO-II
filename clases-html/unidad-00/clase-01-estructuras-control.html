Laboratorio 1: Men√∫ de Usuarios Mejorado (25 min)

Crea un men√∫ interactivo para gestionar usuarios (base del proyecto TaskFlow).

# sistema_usuarios.py
usuarios = []

def mostrar_menu():
    print("\n=== SISTEMA DE USUARIOS ===")
    print("1. Ver usuarios")
    print("2. Agregar usuario")
    print("3. Buscar usuario")
    print("4. Salir")

def ver_usuarios():
    if len(usuarios) == 0:
        print("\nNo hay usuarios registrados.")
        return
    
    print("\n--- Lista de Usuarios ---")
    for i, usuario in enumerate(usuarios, 1):
        print(f"{i}. {usuario}")

def agregar_usuario():
    nombre = input("\nIngresa el nombre del usuario: ")
    
    # Validar que no est√© vac√≠o
    if not nombre.strip():
        print("‚ùå El nombre no puede estar vac√≠o.")
        return
    
    # Validar que no exista
    if nombre in usuarios:
        print("‚ùå Ese usuario ya existe.")
        return
    
    usuarios.append(nombre)
    print(f"‚úÖ Usuario '{nombre}' agregado correctamente.")

def buscar_usuario():
    if len(usuarios) == 0:
        print("\nNo hay usuarios para buscar.")
        return
    
    nombre = input("\n¬øQu√© usuario buscas? ")
    
    for usuario in usuarios:
        if usuario.lower() == nombre.lower():
            print(f"‚úÖ Usuario encontrado: {usuario}")
            return
    
    print(f"‚ùå Usuario '{nombre}' no encontrado.")

# Programa principal
while True:
    mostrar_menu()
    opcion = input("\nSelecciona una opci√≥n (1-4): ")
    
    if opcion == "1":
        ver_usuarios()
    elif opcion == "2":
        agregar_usuario()
    elif opcion == "3":
        buscar_usuario()
    elif opcion == "4":
        print("\n¬°Hasta luego!")
        break
    else:
        print("\n‚ùå Opci√≥n no v√°lida. Intenta de nuevo.")
Conexi√≥n con TaskFlow: Este men√∫ es la base del sistema. M√°s adelante agregaremos: validaciones de email, contrase√±as, estados (activo/inactivo), y persistencia en base de datos.

Retos Adicionales

  • Agrega opci√≥n 5: Eliminar usuario por nombre
  • Valida que el nombre tenga al menos 3 caracteres
  • Permite buscar coincidencias parciales (ej: "ju" encuentra "juan")

Laboratorio 2: Sistema de Calificaciones (25 min)

Crea un sistema completo para gestionar calificaciones de estudiantes con m√∫ltiples evaluaciones.

Requisitos

  • Registrar m√∫ltiples estudiantes
  • Cada estudiante tiene 3 calificaciones (parcial 1, parcial 2, final)
  • Calcular promedio ponderado: P1(30%) + P2(30%) + Final(40%)
  • Determinar si aprob√≥ (promedio >= 3.0)
  • Mostrar estad√≠sticas: mejor estudiante, promedio general, tasa de aprobaci√≥n

C√≥digo Base

# sistema_calificaciones.py
print("üìö SISTEMA DE CALIFICACIONES üìö\n")

estudiantes = []

while True:
    print("\n" + "="*50)
    print("MEN√ö PRINCIPAL")
    print("="*50)
    print("1. Registrar estudiante")
    print("2. Ver todos los estudiantes")
    print("3. Ver estad√≠sticas")
    print("4. Buscar estudiante")
    print("5. Salir")
    
    opcion = input("\nSelecciona una opci√≥n: ")
    
    if opcion == "1":
        nombre = input("Nombre del estudiante: ").strip()
        
        if not nombre:
            print("‚ùå El nombre no puede estar vac√≠o")
            continue
        
        # Verificar si ya existe
        ya_existe = False
        for est in estudiantes:
            if est["nombre"].lower() == nombre.lower():
                ya_existe = True
                break
        
        if ya_existe:
            print("‚ùå Este estudiante ya est√° registrado")
            continue
        
        # Pedir calificaciones con validaci√≥n
        def pedir_nota(mensaje):
            while True:
                try:
                    nota = float(input(mensaje))
                    if 0 <= nota <= 5:
                        return nota
                    print("‚ùå La nota debe estar entre 0 y 5")
                except ValueError:
                    print("‚ùå Ingresa un n√∫mero v√°lido")
        
        p1 = pedir_nota("Parcial 1 (30%): ")
        p2 = pedir_nota("Parcial 2 (30%): ")
        final = pedir_nota("Final (40%): ")
        
        # Calcular promedio ponderado
        promedio = (p1 * 0.3) + (p2 * 0.3) + (final * 0.4)
        
        # Determinar estado
        if promedio >= 3.0:
            estado = "‚úÖ Aprobado"
        else:
            estado = "‚ùå Reprobado"
        
        estudiante = {
            "nombre": nombre,
            "p1": p1,
            "p2": p2,
            "final": final,
            "promedio": promedio,
            "estado": estado
        }
        
        estudiantes.append(estudiante)
        print(f"\n‚úÖ Estudiante registrado. Promedio: {promedio:.2f} - {estado}")
    
    elif opcion == "2":
        if not estudiantes:
            print("\nüì≠ No hay estudiantes registrados")
            continue
        
        print("\n" + "="*70)
        print(f"{'NOMBRE':<20} {'P1':<6} {'P2':<6} {'FINAL':<6} {'PROM':<6} {'ESTADO'}")
        print("="*70)
        
        for est in estudiantes:
            print(f"{est['nombre']:<20} {est['p1']:<6.1f} {est['p2']:<6.1f} "
                  f"{est['final']:<6.1f} {est['promedio']:<6.2f} {est['estado']}")
        print("="*70)
    
    elif opcion == "3":
        if not estudiantes:
            print("\nüì≠ No hay datos para estad√≠sticas")
            continue
        
        # Calcular estad√≠sticas
        total_est = len(estudiantes)
        aprobados = 0
        suma_promedios = 0
        mejor_estudiante = estudiantes[0]
        
        for est in estudiantes:
            suma_promedios += est["promedio"]
            if "Aprobado" in est["estado"]:
                aprobados += 1
            if est["promedio"] > mejor_estudiante["promedio"]:
                mejor_estudiante = est
        
        promedio_general = suma_promedios / total_est
        tasa_aprobacion = (aprobados / total_est) * 100
        
        print("\n" + "="*50)
        print("üìä ESTAD√çSTICAS")
        print("="*50)
        print(f"Total estudiantes: {total_est}")
        print(f"Aprobados: {aprobados}")
        print(f"Reprobados: {total_est - aprobados}")
        print(f"Tasa de aprobaci√≥n: {tasa_aprobacion:.1f}%")
        print(f"Promedio general: {promedio_general:.2f}")
        print(f"\nüèÜ Mejor estudiante: {mejor_estudiante['nombre']}")
        print(f"   Promedio: {mejor_estudiante['promedio']:.2f}")
        print("="*50)
    
    elif opcion == "4":
        if not estudiantes:
            print("\nüì≠ No hay estudiantes registrados")
            continue
        
        buscar = input("Nombre a buscar: ").strip().lower()
        encontrados = []
        
        for est in estudiantes:
            if buscar in est["nombre"].lower():
                encontrados.append(est)
        
        if encontrados:
            print(f"\n‚úÖ {len(encontrados)} estudiante(s) encontrado(s):")
            for est in encontrados:
                print(f"  ‚Ä¢ {est['nombre']}: {est['promedio']:.2f} {est['estado']}")
        else:
            print(f"‚ùå No se encontr√≥ '{buscar}'")
    
    elif opcion == "5":
        print("\nüëã ¬°Hasta luego!")
        break
    
    else:
        print("\n‚ùå Opci√≥n no v√°lida")

Vamos a guardar el sistema de usuarios en Git:

  1. Crea una carpeta clase-01 dentro de poo2-practicas
  2. Copia tu archivo sistema_usuarios.py a esa carpeta
  3. Abre la terminal en VSCode (Ctrl+`)

  5. Agrega el archivo: git add clase-01/sistema_usuarios.py

Los 3 Estados de Git

Estado Descripci√≥n Comando
Working Directory Archivos modificados pero no preparados Editas archivos
Staging Area Archivos listos para guardar git add
Repository Historial guardado git commit

Comandos Esenciales
