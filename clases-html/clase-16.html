Exportar e importar XML

Usamos xml.etree.ElementTree para serializar listas de DTOs a XML y leerlas de vuelta.

import xml.etree.ElementTree as ET

def exportar_xml(estudiantes, archivo):
    root = ET.Element("estudiantes")
    for e in estudiantes:
        nodo = ET.SubElement(root, "estudiante")
        ET.SubElement(nodo, "id").text = str(e["id"])
        ET.SubElement(nodo, "nombre").text = e["nombre"]
        ET.SubElement(nodo, "promedio").text = str(e["promedio"])
    ET.ElementTree(root).write(archivo, encoding="utf-8", xml_declaration=True)

def importar_xml(archivo):
    tree = ET.parse(archivo)
    root = tree.getroot()
    return [
        {
            "id": int(n.find("id").text),
            "nombre": n.find("nombre").text,
            "promedio": float(n.find("promedio").text),
        }
        for n in root.findall("estudiante")
    ]

Data binding en Angular

// estudiantes.component.ts
import { Component, signal } from '@angular/core';

@Component({
  selector: 'app-estudiantes',
  templateUrl: './estudiantes.component.html',
})
export class EstudiantesComponent {
  estudiantes = signal<any[]>([]);
  seleccionado = signal({ id: 0, nombre: '', apellido: '', email: '' });

  seleccionar(est) {
    this.seleccionado.set({ ...est });
  }
}
