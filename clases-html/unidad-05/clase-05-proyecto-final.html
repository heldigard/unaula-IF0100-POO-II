Martes 19/05 Testing de Integración

Los tests de integración verifican que tu API, base de datos y autenticación funcionen juntas.

1. Setup de Tests con SQLite en memoria

# tests/conftest.py
import pytest
from fastapi.testclient import TestClient
from sqlalchemy import create_engine
from sqlalchemy.orm import sessionmaker
from app.main import app
from app.infrastructure.database import Base, get_db

TEST_DATABASE_URL = "sqlite:///./test.db"
engine = create_engine(TEST_DATABASE_URL, connect_args={"check_same_thread": False})
TestingSessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def override_get_db():
    db = TestingSessionLocal()
    try:
        yield db
    finally:
        db.close()

app.dependency_overrides[get_db] = override_get_db

@pytest.fixture(scope="function")
def client():
    Base.metadata.create_all(bind=engine)
    yield TestClient(app)
    Base.metadata.drop_all(bind=engine)

2. Test de Registro y Login

# tests/test_auth.py
def test_registro_y_login(client):
    # Crear usuario
    response = client.post("/api/v1/usuarios", json={
        "username": "testuser",
        "email": "test@test.com",
        "password": "Test1234"
    })
    assert response.status_code == 201

    # Login
    response = client.post("/api/v1/auth/login", data={
        "username": "testuser",
        "password": "Test1234"
    })
    assert response.status_code == 200
    assert "access_token" in response.json()

import pytest

@pytest.fixture
def auth_token(client):
    client.post("/api/v1/usuarios", json={
        "username": "demo",
        "email": "demo@test.com",
        "password": "Demo1234"
    })
    response = client.post("/api/v1/auth/login", data={
        "username": "demo",
        "password": "Demo1234"
    })
    return response.json()["access_token"]

def test_crear_tarea_autenticado(client, auth_token):
    headers = {"Authorization": f"Bearer {auth_token}"}
    response = client.post("/api/v1/tareas", json={
        "titulo": "Tarea test",
        "descripcion": "Desc",
        "prioridad": "media",
        "proyecto_id": 1
    }, headers=headers)
    assert response.status_code == 201
    data = response.json()
    assert data["titulo"] == "Tarea test"

def test_listar_tareas_sin_token(client):
    response = client.get("/api/v1/tareas")
    assert response.status_code == 401

Jueves 21/05 Pulido del Proyecto

Dia dedicado a revisar detalles que marcan la diferencia entre un proyecto funcional y uno profesional.

Checklist de Pulido

  • Todos los endpoints responden con status codes correctos (201, 404, 422, 401)
  • JWT protege las rutas que deben estar protegidas
  • Passwords nunca se devuelven en responses
  • Paginación funciona en listados grandes
  • Filtros de búsqueda no crashean con strings vacíos
  • Manejo de errores devuelve JSON consistente
  • Relaciones cargan sin N+1 (usa joinedload donde sea necesario)
  • README tiene instrucciones claras de instalación y uso
  • Colección Postman/Insomnia incluida en repo

Ejemplo: Response consistente de errores

# api/errors.py
from fastapi import Request
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError

async def validation_handler(request: Request, exc: RequestValidationError):
    return JSONResponse(
        status_code=422,
        content={
            "error": True,
            "mensaje": "Datos de entrada invalidos",
            "detalle": exc.errors()
        }
    )

# En main.py
app.add_exception_handler(RequestValidationError, validation_handler)
