import math

class Fraccion:
    def __init__(self, numerador, denominador):
        if denominador == 0:
            raise ValueError("El denominador no puede ser 0")
        self.numerador = numerador
        self.denominador = denominador

    # Sobrecarga del operador +
    def __add__(self, other):
        num = (self.numerador * other.denominador +
               other.numerador * self.denominador)
        den = self.denominador * other.denominador
        return Fraccion(num, den)

    # Sobrecarga del operador ==
    def __eq__(self, other):
        return (self.numerador * other.denominador ==
                other.numerador * self.denominador)

    def __str__(self):
        return f"{self.numerador}/{self.denominador}"

# Uso
f1 = Fraccion(1, 2)
f2 = Fraccion(1, 4)
f3 = f1 + f2  # Usa __add__
print(f3)  # "6/8"

class Calculadora:
    # Constructor con valores por defecto
    def __init__(self, marca="Gen√©rica"):
        self.marca = marca

    # 2 o 3 enteros usando valores por defecto
    def sumar(self, a, b, c=None):
        if c is None:
            return a + b
        return a + b + c

    # 2 floats
    def sumar_float(self, a, b):
        return a + b

    # N enteros (*args)
    def sumar_n(self, *numeros):
        return sum(numeros)

# Uso

print(calc.sumar_float(5.5, 3.2)) # 8.7 (float, float)
print(calc.sumar_n(1, 2, 3, 4))  # 10 (*args)
