from behave import given, when, then
from estudiante import Estudiante

@given('el estudiante con codigo "{codigo}"')
def step_estudiante_codigo(context, codigo):
    context.estudiante = Estudiante(codigo=codigo)

@when('registro las notas')
def step_registro_notas(context):
    for fila in context.table:
        parcial = int(fila['Parcial'])
        nota = float(fila['Nota'])
        context.estudiante.registrar_nota(parcial, nota)
    context.promedio = context.estudiante.calcular_promedio()

@then('el promedio debe ser {esperado:f}')
def step_promedio_esperado(context, esperado):
    assert abs(context.promedio - esperado) < 0.01,         f"Esperado {esperado}, obtenido {context.promedio}"
