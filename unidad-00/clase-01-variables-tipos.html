Conceptos Te√≥ricos

Tipos de Datos B√°sicos en Python

Python tiene varios tipos de datos incorporados. Los m√°s fundamentales son:

1. Enteros (int)

Numeros sin parte decimal. Pueden ser positivos, negativos o cero.

edad: int = 25
cantidad: int = -100
numero_grande: int = 1_000_000  # Python 3.6+ permite guiones para legibilidad

2. Flotantes (float)

Numeros con parte decimal. Nota: los flotantes tienen precision limitada.

precio: float = 19.99
pi: float = 3.14159
notacion_cientifica: float = 1.5e-4  # 0.00015
‚ö†Ô∏è Precisi√≥n de flotantes: 0.1 + 0.2 == 0.3 retorna False en Python debido a c√≥mo se almacenan los flotantes. Para c√°lculos monetarios, considera usar decimal.Decimal.

3. Strings (str)

Secuencias inmutables de caracteres. Pueden usar comillas simples, dobles o triples.

nombre: str = "Ana"          # Comillas dobles
saludo: str = 'Hola'        # Comillas simples
parrafo: str = """Linea 1
Linea 2"""            # Comillas triples para multiples lineas

4. Booleanos (bool)

Representan valores de verdad. Solo dos valores: True o False.

es_activo: bool = True
esta_completado: bool = False
es_mayor: bool = (5 > 3)  # True

5. NoneType (None)

Representa la ausencia de valor o un valor nulo.

resultado: None = None
valor_inicial: None = None

Operadores de Comparaci√≥n

Los operadores de comparaci√≥n retornan siempre un valor booleano (True o False):

Operador Descripci√≥n Ejemplo Resultado
== Igual a 5 == 5 True
!= Diferente de 5 != 3 True
< Menor que 3 < 5 True
> Mayor que 5 > 3 True
<= Menor o igual que 5 <= 5 True
>= Mayor o igual que 5 >= 3 True

Operadores L√≥gicos

Usados para combinar expresiones booleanas:

Operador Descripci√≥n Ejemplo Resultado
and Retorna True si ambos son True True and False False
or Retorna True si al menos uno es True True or False True
not Invierte el valor booleano not True False

Conversi√≥n de Tipos (Type Casting)

Python permite convertir expl√≠citamente entre tipos usando funciones integradas:

# A string

# A booleano (casi todo es True excepto valores "vac√≠os")

b3 = bool("hola")     # True
b4 = bool("")         # False

‚ùå Errores comunes:

Vamos a crear una calculadora que realice operaciones b√°sicas con validaci√≥n de entrada y manejo de errores.

Paso 1: Estructura B√°sica

# calculadora.py
"""
Calculadora con validaciones
Permite realizar operaciones b√°sicas con dos n√∫meros.
"""

def mostrar_menu() -> None:
    """Muestra el men√∫ de operaciones."""
    print("\n=== Calculadora ===")
    print("1. Suma")
    print("2. Resta")
    print("3. Multiplicaci√≥n")
    print("4. Divisi√≥n")
    print("5. Salir")

def obtener_numero(prompt: str) -> float:
    """Solicita un n√∫mero al usuario con validaci√≥n."""
    while True:
        try:
            valor = input(prompt)
            return float(valor)
        except ValueError:
            print("‚ùå Error: Debe ingresar un n√∫mero v√°lido.")

def realizar_operacion(opcion: int, num1: float, num2: float) -> float | None:
    """Realiza la operaci√≥n seleccionada."""
    match opcion:
        case 1:
            resultado = num1 + num2
            print(f"‚úÖ {num1} + {num2} = {resultado}")
        case 2:
            resultado = num1 - num2
            print(f"‚úÖ {num1} - {num2} = {resultado}")
        case 3:
            resultado = num1 * num2
            print(f"‚úÖ {num1} √ó {num2} = {resultado}")
        case 4:
            if num2 == 0:
                print("‚ùå Error: No se puede dividir por cero.")
                return None
            resultado = num1 / num2
            print(f"‚úÖ {num1} √∑ {num2} = {resultado:.4f}")
        case _:
            print("‚ùå Opci√≥n no v√°lida.")
            return None

    return resultado

def main() -> None:
    """Funci√≥n principal de la calculadora."""
    historial = []

    while True:
        mostrar_menu()

        # Obtener opci√≥n del men√∫
        opcion = input("Selecciona una operaci√≥n (1-5): ")
        try:
            opcion = int(opcion)
        except ValueError:
            print("‚ùå Error: Debe ingresar un n√∫mero del 1 al 5.")
            continue

        # Salir si el usuario lo solicita
        if opcion == 5:
            print("\n¬°Gracias por usar la calculadora!")
            break

        # Obtener n√∫meros
        num1 = obtener_numero("Primer n√∫mero: ")
        num2 = obtener_numero("Segundo n√∫mero: ")

        # Realizar operaci√≥n
        resultado = realizar_operacion(opcion, num1, num2)

        # Guardar en historial si la operaci√≥n fue exitosa
        if resultado is not None:
            operacion = {
                "opcion": opcion,
                "num1": num1,
                "num2": num2,
                "resultado": resultado
            }
            historial.append(operacion)

            # Mostrar historial
            if len(historial) > 0:
                print(f"\nüìä Historial ({len(historial)} operaciones)")
                for i, op in enumerate(historial[-3:], 1):  # √öltimas 3
                    simbolos = {1: "+", 2: "-", 3: "√ó", 4: "√∑"}
                    simbolo = simbolos.get(op["opcion"], "?")
                    print(f"  {i}. {op['num1']} {simbolo} {op['num2']} = {op['resultado']}")

if __name__ == "__main__":
    main()

# Modelo simplificado de Usuario para TaskFlow
class Usuario:
    def __init__(
        self,
        username: str,
        email: str,
        edad: int | None = None,
        activo: bool = True
    ):
        self.username = username  # str
        self.email = email        # str
        self.edad = edad          # int | None
        self.activo = activo      # bool

    def validar(self) -> list[str]:
        """Valida los campos del usuario."""
        errores = []

        # Validar username: str con longitud m√≠nima
        if not isinstance(self.username, str):
            errores.append("Username debe ser string")
        elif len(self.username) < 3:
            errores.append("Username debe tener al menos 3 caracteres")

        # Validar email: str con @
        if not isinstance(self.email, str):
            errores.append("Email debe ser string")
        elif "@" not in self.email:
            errores.append("Email debe contener @")

        # Validar edad: int positivo si no es None
        if self.edad is not None:
            if not isinstance(self.edad, int):
                errores.append("Edad debe ser entero")
            elif self.edad < 0:
                errores.append("Edad no puede ser negativa")

        # Validar activo: bool
        if not isinstance(self.activo, bool):
            errores.append("Activo debe ser booleano")

        return errores

    def es_valido(self) -> bool:
        """Verifica si el usuario es v√°lido."""
        return len(self.validar()) == 0

# Ejemplo de uso
usuario = Usuario("juan123", "juan@example.com", 25, True)
if usuario.es_valido():
    print(f"‚úÖ Usuario {usuario.username} es v√°lido")
else:
    print("‚ùå Errores:", usuario.validar())
