Ejercicio: Gestor de Contactos (10 min)

Sistema completo usando listas y diccionarios (base para TaskFlow).

# contactos.py
contactos = []

def agregar_contacto(nombre, telefono, email):
    """Agrega un contacto nuevo."""
    contacto = {
        "nombre": nombre,
        "telefono": telefono,
        "email": email
    }
    contactos.append(contacto)
    print(f"‚úÖ Contacto '{nombre}' agregado.")

def mostrar_contactos():
    """Muestra todos los contactos."""
    if not contactos:
        print("No hay contactos.")
        return
    
    print("\n--- Contactos ---")
    for i, c in enumerate(contactos, 1):
        print(f"{i}. {c['nombre']}")
        print(f"   Tel: {c['telefono']}")
        print(f"   Email: {c['email']}")

def buscar_contacto(nombre):
    """Busca contactos por nombre."""
    encontrados = []
    for c in contactos:
        if nombre.lower() in c["nombre"].lower():
            encontrados.append(c)
    
    if encontrados:
        print(f"\nüîç {len(encontrados)} encontrado(s):")
        for c in encontrados:
            print(f"   - {c['nombre']}")
    else:
        print(f"‚ùå No se encontr√≥ '{nombre}'")

# Programa principal
while True:
    print("\n=== GESTOR DE CONTACTOS ===")
    print("1. Agregar contacto")
    print("2. Ver contactos")
    print("3. Buscar contacto")
    print("4. Salir")
    
    opcion = input("\nOpci√≥n: ")
    
    if opcion == "1":
        nombre = input("Nombre: ")
        telefono = input("Tel√©fono: ")
        email = input("Email: ")
        agregar_contacto(nombre, telefono, email)
    elif opcion == "2":
        mostrar_contactos()
    elif opcion == "3":
        nombre = input("Nombre a buscar: ")
        buscar_contacto(nombre)
    elif opcion == "4":
        print("¬°Hasta luego!")
        break
    else:
        print("‚ùå Opci√≥n inv√°lida")
