# contactos.py
contactos = []
categorias_validas = {"amigo", "familia", "trabajo", "otro"}

def validar_email(email):
    """Valida que el email contenga @ y ."""
    return "@" in email and "." in email

def agregar_contacto(nombre, telefono, email, categoria="otro"):
    """Agrega un contacto nuevo con validaciones."""
    # Validaciones
    if not nombre.strip():
        print("‚ùå El nombre no puede estar vac√≠o")
        return False
    
    if not validar_email(email):
        print("‚ùå Email inv√°lido. Debe contener @ y punto")
        return False
    
    if categoria.lower() not in categorias_validas:
        print(f"‚ùå Categor√≠a inv√°lida. Opciones: {', '.join(categorias_validas)}")
        return False
    
    # Verificar duplicado por email
    for c in contactos:
        if c["email"].lower() == email.lower():
            print("‚ùå Ya existe un contacto con ese email")
            return False
    
    contacto = {
        "nombre": nombre,
        "telefono": telefono,
        "email": email,
        "categoria": categoria.lower()
    }
    contactos.append(contacto)
    print(f"‚úÖ Contacto '{nombre}' agregado correctamente.")
    return True

def mostrar_contactos(filtrar_categoria=None):
    """Muestra contactos, opcionalmente filtrados por categor√≠a."""
    if not contactos:
        print("üì≠ No hay contactos registrados.")
        return
    
    contactos_mostrar = contactos
    if filtrar_categoria:
        contactos_mostrar = [c for c in contactos if c["categoria"] == filtrar_categoria]
    
    if not contactos_mostrar:
        print(f"üì≠ No hay contactos en la categor√≠a '{filtrar_categoria}'")
        return
    
    print("\n" + "="*60)
    print(f"{'N¬∞':<4} {'NOMBRE':<20} {'TEL√âFONO':<15} {'CATEGOR√çA'}")
    print("="*60)
    
    for i, c in enumerate(contactos_mostrar, 1):
        print(f"{i:<4} {c['nombre']:<20} {c['telefono']:<15} {c['categoria']}")
        print(f"     üìß {c['email']}")
    print("="*60)
    print(f"Total: {len(contactos_mostrar)} contacto(s)")

def buscar_contacto(termino):
    """Busca contactos por nombre, tel√©fono o email."""
    if not contactos:
        print("üì≠ No hay contactos registrados")
        return
    
    termino = termino.lower()
    encontrados = []
    
    for c in contactos:
        if (termino in c["nombre"].lower() or 
            termino in c["telefono"] or 
            termino in c["email"].lower()):
            encontrados.append(c)
    
    if encontrados:
        print(f"\nüîç {len(encontrados)} resultado(s) para '{termino}':")
        for c in encontrados:
            print(f"  ‚Ä¢ {c['nombre']} ({c['categoria']})")
    else:
        print(f"‚ùå No se encontraron resultados para '{termino}'")

def eliminar_contacto(email):
    """Elimina un contacto por email."""
    for i, c in enumerate(contactos):
        if c["email"].lower() == email.lower():
            confirmar = input(f"¬øEliminar a {c['nombre']}? (s/n): ")
            if confirmar.lower() == 's':
                contactos.pop(i)
                print(f"‚úÖ Contacto '{c['nombre']}' eliminado")
                return True
            else:
                print("Cancelado")
                return False
    print(f"‚ùå No se encontr√≥ contacto con email '{email}'")
    return False

def mostrar_estadisticas():
    """Muestra estad√≠sticas de los contactos."""
    if not contactos:
        print("üì≠ No hay datos para estad√≠sticas")
        return
    
    total = len(contactos)
    print("\n" + "="*40)
    print("üìä ESTAD√çSTICAS")
    print("="*40)
    print(f"Total de contactos: {total}")
    
    # Contar por categor√≠a
    conteo = {}
    for c in contactos:
        cat = c["categoria"]
        conteo[cat] = conteo.get(cat, 0) + 1
    
    print("\nPor categor√≠a:")
    for cat, cantidad in sorted(conteo.items()):
        porcentaje = (cantidad / total) * 100
        print(f"  ‚Ä¢ {cat.capitalize():<12}: {cantidad:>2} ({porcentaje:.1f}%)")
    print("="*40)

# Programa principal
while True:

C√≥digo Base

# inventario.py
print("üì¶ SISTEMA DE INVENTARIO üì¶\n")

inventario = []
id_contador = 1

def generar_id():
    """Genera un ID √∫nico para cada producto."""
    global id_contador
    id_actual = f"PROD-{id_contador:04d}"
    id_contador += 1
    return id_actual

def agregar_producto(nombre, categoria, precio, cantidad):
    """Agrega un nuevo producto al inventario."""
    if precio <= 0 or cantidad < 0:
        print("‚ùå Precio y cantidad deben ser positivos")
        return None
    
    producto = {
        "id": generar_id(),
        "nombre": nombre,
        "categoria": categoria,
        "precio": precio,
        "cantidad": cantidad
    }
    inventario.append(producto)
    print(f"‚úÖ Producto '{nombre}' agregado con ID: {producto['id']}")
    return producto

def calcular_valor_inventario():
    """Calcula el valor total del inventario."""
    total = 0
    for p in inventario:
        total += p["precio"] * p["cantidad"]
    return total

def productos_bajo_stock(limite=5):
    """Retorna lista de productos con stock bajo."""
    return [p for p in inventario if p["cantidad"] < limite]

def buscar_producto(termino):
    """Busca productos por nombre o categor√≠a."""
    termino = termino.lower()
    resultados = []
    for p in inventario:
        if termino in p["nombre"].lower() or termino in p["categoria"].lower():
            resultados.append(p)
    return resultados

def actualizar_stock(id_producto, cantidad, es_entrada=True):
    """Actualiza el stock de un producto."""
    for p in inventario:
        if p["id"] == id_producto:
            if es_entrada:
                p["cantidad"] += cantidad
                print(f"‚úÖ Entrada: {cantidad} unidades de {p['nombre']}")
            else:
                if p["cantidad"] >= cantidad:
                    p["cantidad"] -= cantidad
                    print(f"‚úÖ Salida: {cantidad} unidades de {p['nombre']}")
                else:
                    print(f"‚ùå Stock insuficiente. Disponible: {p['cantidad']}")
            return True
    print(f"‚ùå Producto con ID '{id_producto}' no encontrado")
    return False

def mostrar_inventario():
    """Muestra todo el inventario en formato tabla."""
    if not inventario:
        print("üì≠ Inventario vac√≠o")
        return
    
    print("\n" + "="*90)
    print(f"{'ID':<12} {'NOMBRE':<20} {'CATEGOR√çA':<15} {'PRECIO':<10} {'CANT':<6} {'VALOR'}")
    print("="*90)
    
    for p in inventario:
        valor = p["precio"] * p["cantidad"]
        print(f"{p['id']:<12} {p['nombre']:<20} {p['categoria']:<15} "
              f"${p['precio']:<9.2f} {p['cantidad']:<6} ${valor:.2f}")
    
    print("="*90)
    print(f"üí∞ Valor total del inventario: ${calcular_valor_inventario():.2f}")

def mostrar_alertas_stock():
    """Muestra productos con stock bajo."""
    bajos = productos_bajo_stock()

def resumen_por_categoria():
    """Muestra resumen agrupado por categor√≠a."""
    if not inventario:
        return
    
    categorias = {}
    for p in inventario:
        cat = p["categoria"]
        if cat not in categorias:
            categorias[cat] = {"cantidad": 0, "valor": 0}
        categorias[cat]["cantidad"] += p["cantidad"]
        categorias[cat]["valor"] += p["precio"] * p["cantidad"]
    
    print("\nüìä RESUMEN POR CATEGOR√çA:")
    print("="*60)
    for cat, datos in sorted(categorias.items()):
        print(f"{cat.capitalize():<20}: {datos['cantidad']:>4} unidades - ${datos['valor']:.2f}")
    print("="*60)
