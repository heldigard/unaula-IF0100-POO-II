Condicionales: if, elif, else

Los condicionales permiten que tu programa tome decisiones basadas en condiciones.

Sintaxis B√°sica

if condicion:
    # c√≥digo si condici√≥n es True
    pass

    pass

Manejo de Excepciones

Las excepciones permiten manejar errores sin que el programa se cierre.

Sintaxis B√°sica

try:
    # C√≥digo que podr√≠a causar error
    resultado = 10 / 0
except ZeroDivisionError:
    # Manejo del error espec√≠fico
    print("No se puede dividir por cero")
except Exception as e:
    # Manejo de cualquier otro error
    print(f"Error: {e}")
else:
    # Se ejecuta si no hay errores
    print("Operaci√≥n exitosa")
finally:
    # Se ejecuta siempre, haya error o no
    print("Limpieza final")

def validar_password(password: str) -> tuple[bool, list[str]]:
    """
    Valida una contrase√±a seg√∫n criterios espec√≠ficos.
    Retorna (es_valida, lista_errores)
    """
    errores = []

    # Validar longitud
    if len(password) < 8:
        errores.append("La contrase√±a debe tener al menos 8 caracteres")

    # Validar may√∫scula
    tiene_mayuscula = False
    for caracter in password:
        if caracter.isupper():
            tiene_mayuscula = True
            break

    if not tiene_mayuscula:
        errores.append("La contrase√±a debe tener al menos una may√∫scula")

    # Validar min√∫scula
    tiene_minuscula = False
    for caracter in password:
        if caracter.islower():
            tiene_minuscula = True
            break

    if not tiene_minuscula:
        errores.append("La contrase√±a debe tener al menos una min√∫scula")

    # Validar n√∫mero
    tiene_numero = False
    for caracter in password:
        if caracter.isdigit():
            tiene_numero = True
            break

    if not tiene_numero:
        errores.append("La contrase√±a debe tener al menos un n√∫mero")

    return len(errores) == 0, errores

# Uso

# adivinanza.py
"""
Juego de Adivinanza
El usuario debe adivinar un n√∫mero entre 1 y 100.
Tiene 10 intentos.
"""

import random

def obtener_intento(intentos_restantes: int) -> int:
    """Solicita un n√∫mero al usuario con validaci√≥n."""
    while True:
        try:
            valor = input(f"\nIntento {10 - intentos_restantes + 1}/10. Ingresa n√∫mero (1-100): ")
            numero = int(valor)
            if numero < 1 or numero > 100:
                print("‚ùå El n√∫mero debe estar entre 1 y 100.")
                continue
            return numero
        except ValueError:
            print("‚ùå Debe ingresar un n√∫mero entero.")

def jugar() -> None:
    """Funci√≥n principal del juego."""
    # Generar n√∫mero aleatorio
    numero_secreto = random.randint(1, 100)
    print("=== Juego de Adivinanza ===")

    print("¬°Tienes 10 intentos para adivinarlo!")

        # Obtener intento del usuario
        numero_usuario = obtener_intento(intento)

        # Verificar si adivin√≥
        if numero_usuario == numero_secreto:
            print(f"\nüéâ ¬°Felicidades! Adivinaste el n√∫mero en {intento} intento(s).")
            return

        # Dar pista
        if numero_usuario < numero_secreto:
            print("üìà El n√∫mero es MAYOR")
        else:
            print("üìâ El n√∫mero es MENOR")

        # Mostrar n√∫meros usados
        print(f"N√∫meros usados: {intento}/10")

    # Si se acabaron los intentos
    print(f"\nüò¢ Se acabaron los intentos. El n√∫mero era {numero_secreto}.")

def jugar_de_nuevo() -> bool:
    """Pregunta al usuario si quiere jugar de nuevo."""

        elif respuesta in ["n", "no"]:

        else:

        if not jugar_de_nuevo():

# Modelo Usuario con validaciones
from enum import Enum

class EstadoUsuario(Enum):
    PENDIENTE = "pendiente"
    ACTIVO = "activo"
    BLOQUEADO = "bloqueado"

class Usuario:
    def __init__(
        self,
        username: str,
        email: str,
        password: str,
        estado: EstadoUsuario = EstadoUsuario.PENDIENTE
    ):
        self.username = username
        self.email = email
        self.password = password
        self.estado = estado

    def validar(self) -> list[str]:
        """Valida todos los campos del usuario."""
        errores = []

        # Validar username
        if not self.username:
            errores.append("Username es requerido")
        elif len(self.username) < 3:
            errores.append("Username debe tener al menos 3 caracteres")
        elif len(self.username) > 50:
            errores.append("Username no puede exceder 50 caracteres")
        else:
            # Verificar que solo contenga caracteres v√°lidos
            for caracter in self.username:
                if not (caracter.isalnum() or caracter == "_"):
                    errores.append("Username solo puede contener letras, n√∫meros y _")
                    break

        # Validar email
        if not self.email:
            errores.append("Email es requerido")
        elif "@" not in self.email:
            errores.append("Email debe contener @")
        else:
            # Validaciones adicionales de email
            partes = self.email.split("@")
            if len(partes) != 2:
                errores.append("Email inv√°lido")
            elif "." not in partes[1]:
                errores.append("Email debe tener dominio v√°lido")

        # Validar password
        if not self.password:
            errores.append("Password es requerido")
        elif len(self.password) < 8:
            errores.append("Password debe tener al menos 8 caracteres")
        else:
            # Validar complejidad del password
            tiene_mayuscula = False
            tiene_minuscula = False
            tiene_numero = False

            for caracter in self.password:
                if caracter.isupper():
                    tiene_mayuscula = True
                elif caracter.islower():
                    tiene_minuscula = True
                elif caracter.isdigit():
                    tiene_numero = True

            if not tiene_mayuscula:
                errores.append("Password debe tener al menos una may√∫scula")
            if not tiene_minuscula:
                errores.append("Password debe tener al menos una min√∫scula")
            if not tiene_numero:
                errores.append("Password debe tener al menos un n√∫mero")

        # Validar estado
        estados_validos = [EstadoUsuario.PENDIENTE, EstadoUsuario.ACTIVO, EstadoUsuario.BLOQUEADO]
        if self.estado not in estados_validos:
            errores.append("Estado no v√°lido")

        return errores

    def es_valido(self) -> bool:
        """Verifica si el usuario es v√°lido."""
        return len(self.validar()) == 0

# Uso
usuario = Usuario(
    username="juan123",
    email="juan@example.com",
    password="MiPassword123",
    estado=EstadoUsuario.ACTIVO
)

if usuario.es_valido():
    print(f"‚úÖ Usuario '{usuario.username}' es v√°lido")
else:
    print("‚ùå Errores de validaci√≥n:")
    for error in usuario.validar():
        print(f"  - {error}")
